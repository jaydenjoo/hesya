```sql
-- Phase 1-β: AI 초안 검수·승인 모드 + Bot 자동 토글
--
-- spec: docs/superpowers/specs/2026-05-07-phase-1-beta-design.md §5.2
--
-- 변경:
--   - stores.bot_mode (boolean) — false=owner 검수·승인, true=AI 자동
--   - messages.draft_status (text) — NULL/pending_review/approved/sent/skipped/direct
--   - messages.reviewed_by (uuid) — owner 처리 actor
--   - messages.edited_from_ai (boolean) — H1 수정률 분석용
--
-- RLS: 신규 정책 없음. 기존 stores/messages 정책 그대로 적용. service_role
--      bypass 유지.
--
-- ROLLBACK:
--   ALTER TABLE messages DROP CONSTRAINT messages_draft_status_check;
--   ALTER TABLE messages DROP COLUMN edited_from_ai;
--   ALTER TABLE messages DROP COLUMN reviewed_by;
--   ALTER TABLE messages DROP COLUMN draft_status;
--   ALTER TABLE stores DROP COLUMN bot_mode;

ALTER TABLE stores
  ADD COLUMN bot_mode boolean NOT NULL DEFAULT false;

COMMENT ON COLUMN stores.bot_mode IS
  'false=owner 검수·승인, true=AI 자동 전송. Phase 1-β 학습 가설 H1 토글.';

ALTER TABLE messages
  ADD COLUMN draft_status text,
  ADD COLUMN reviewed_by uuid,
  ADD COLUMN edited_from_ai boolean;

ALTER TABLE messages
  ADD CONSTRAINT messages_draft_status_check
  CHECK (draft_status IS NULL
    OR draft_status IN ('pending_review','approved','sent','skipped','direct'));

COMMENT ON COLUMN messages.draft_status IS
  'NULL=수신 메시지. pending_review=AI 초안 대기. approved/sent/skipped=owner 처리. direct=owner 직접 작성(Phase 1-β 후 합류).';
COMMENT ON COLUMN messages.edited_from_ai IS
  'AI 초안과 최종 전송 텍스트가 다른지. H1 수정률 분석용.';
```

```typescript
import { eq } from "drizzle-orm";
import { stores, storeVerifications } from "@hesya/database";
import { db } from "@hesya/database";

export async function getStoreBotMode(storeId: string): Promise<boolean> {
  const [row] = await db
    .select({ botMode: stores.botMode })
    .from(stores)
    .where(eq(stores.id, storeId));
  return row?.botMode ?? false;
}

export async function setStoreBotMode(
  storeId: string,
  value: boolean,
): Promise<void> {
  await db.update(stores).set({ botMode: value }).where(eq(stores.id, storeId));
}

export async function listStoresPendingReview() {
  return db
    .select()
    .from(stores)
    .where(eq(stores.verificationStatus, "manual_review"));
}

export async function approveStore(input: {
  storeId: string;
  verificationId: string;
  reviewerId: string;
}): Promise<void> {
  await db.transaction(async (tx) => {
    await tx
      .update(stores)
      .set({ verificationStatus: "auto_approved" })
      .where(eq(stores.id, input.storeId));
    await tx
      .update(storeVerifications)
      .set({
        verificationStatus: "auto_approved",
        reviewedBy: input.reviewerId,
        reviewedAt: new Date(),
      })
      .where(eq(storeVerifications.id, input.verificationId));
  });
}

export async function rejectStore(input: {
  storeId: string;
  verificationId: string;
  reviewerId: string;
  reason: string;
}): Promise<void> {
  await db.transaction(async (tx) => {
    await tx
      .update(stores)
      .set({ verificationStatus: "rejected" })
      .where(eq(stores.id, input.storeId));
    await tx
      .update(storeVerifications)
      .set({
        verificationStatus: "rejected",
        reviewedBy: input.reviewerId,
        reviewedAt: new Date(),
        rejectionReason: input.reason,
      })
      .where(eq(storeVerifications.id, input.verificationId));
  });
}
```

```typescript
"use server";

import { db, storeOwners, storeVerifications, stores } from "@hesya/database";
import { KycApplicationSchema } from "../schema";
import * as Sentry from "@sentry/nextjs";

export type SubmitKycApplicationResult =
  | { ok: true; storeId: string; verificationId: string }
  | { ok: false; error: "validation" | "internal"; message: string };

export async function submitKycApplication(args: {
  userId: string;
  input: unknown;
}): Promise<SubmitKycApplicationResult> {
  const parsed = KycApplicationSchema.safeParse(args.input);
  if (!parsed.success) {
    return { ok: false, error: "validation", message: parsed.error.message };
  }
  const data = parsed.data;
  try {
    return await db.transaction(async (tx) => {
      const [store] = await tx
        .insert(stores)
        .values({
          name: data.storeName,
          phone: data.phone,
          address: { full: data.address },
          businessLicenseNumber: data.businessNumber,
          businessLicenseImageUrl: data.businessLicenseImageUrl,
          verificationStatus: "manual_review",
        })
        .returning({ id: stores.id });
      const [verification] = await tx
        .insert(storeVerifications)
        .values({
          storeId: store.id,
          businessNumber: data.businessNumber,
          representativeName: data.representativeName,
          declarationNoMassage: data.declarationNoMassage,
          declarationNoMedicalDevice: data.declarationNoMedicalDevice,
          declarationNoOrientalMedicine: data.declarationNoOrientalMedicine,
          selfDeclarationSignedAt: new Date(),
          verificationStatus: "manual_review",
        })
        .returning({ id: storeVerifications.id });
      await tx.insert(storeOwners).values({
        userId: args.userId,
        storeId: store.id,
        role: "owner",
      });
      return {
        ok: true as const,
        storeId: store.id,
        verificationId: verification.id,
      };
    });
  } catch (err) {
    Sentry.captureException(err, {
      tags: { phase: "onboarding:submit-kyc" },
      extra: { userId: args.userId },
    });
    return { ok: false, error: "internal", message: "신청 저장 실패" };
  }
}
```

```typescript
import { NextResponse } from "next/server";
import { auth } from "@hesya/auth";
import { headers } from "next/headers";
import { db, storeOwners, stores } from "@hesya/database";
import { eq } from "drizzle-orm";

export async function GET() {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session?.user) {
    return NextResponse.json(
      { ok: false, error: "unauthorized" },
      { status: 401 },
    );
  }
  const owners = await db
    .select({ storeId: storeOwners.storeId })
    .from(storeOwners)
    .where(eq(storeOwners.userId, session.user.id))
    .limit(1);
  if (owners.length === 0) {
    return NextResponse.json({ ok: true, status: "no_application" });
  }
  const [store] = await db
    .select({ verificationStatus: stores.verificationStatus })
    .from(stores)
    .where(eq(stores.id, owners[0].storeId));
  return NextResponse.json({
    ok: true,
    status: store.verificationStatus,
    storeId: owners[0].storeId,
  });
}
```

```tsx
"use client";

import { useEffect, useState } from "react";

type Status = "manual_review" | "auto_approved" | "rejected" | "pending";
type Props = { initialStatus: Status; pollMs?: number };

export function PendingStatus({ initialStatus, pollMs = 30000 }: Props) {
  const [status, setStatus] = useState<Status>(initialStatus);
  useEffect(() => {
    if (status === "auto_approved" || status === "rejected") return;
    const id = setInterval(async () => {
      const res = await fetch("/api/store/me/status");
      const json = await res.json();
      if (json.ok && json.status) setStatus(json.status);
    }, pollMs);
    return () => clearInterval(id);
  }, [status, pollMs]);

  return (
    <div className="max-w-lg">
      {status === "manual_review" && (
        <p>검토 중 — 24~48시간 내 결과를 알려드립니다.</p>
      )}
      {status === "auto_approved" && (
        <p>
          승인됨!{" "}
          <a href="/store/inbox/connect" className="underline">
            Instagram 연결
          </a>
          으로 진행하세요.
        </p>
      )}
      {status === "rejected" && (
        <p>거절됨. 사유는 메일 또는 상담으로 안내드립니다.</p>
      )}
      {status === "pending" && <p>대기 중...</p>}
    </div>
  );
}
```

```typescript
"use server";

import { auth } from "@hesya/auth";
import { headers } from "next/headers";
import { findByUserId } from "@/shared/lib/dal/store-owners";
import { findMessageById } from "@/shared/lib/dal/messages";
import { updateDraftStatus } from "@/shared/lib/dal/messages";
import { sendOutboundMessage } from "@/features/inbox/actions/send-outbound";
import { revalidatePath } from "next/cache";

export type ApproveDraftResult = { ok: true } | { ok: false; error: string };

export async function approveDraft(args: {
  messageId: string;
}): Promise<ApproveDraftResult> {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session?.user) return { ok: false, error: "unauthorized" };
  const owners = await findByUserId(session.user.id);
  const msg = await findMessageById(args.messageId);
  if (!msg) return { ok: false, error: "not_found" };
  if (!owners.find((o) => o.storeId === msg.storeId))
    return { ok: false, error: "forbidden" };
  if (msg.draftStatus !== "pending_review")
    return { ok: false, error: "invalid_state" };
  await updateDraftStatus({
    messageId: msg.id,
    nextStatus: "approved",
    reviewerId: session.user.id,
  });
  // 즉시 전송
  const sendResult = await sendOutboundMessage({ messageId: msg.id });
  if (!sendResult.ok) return { ok: false, error: "send_failed" };
  await updateDraftStatus({
    messageId: msg.id,
    nextStatus: "sent",
    reviewerId: session.user.id,
  });
  revalidatePath("/store/inbox");
  return { ok: true };
}
```

```typescript
import { z } from "zod";

const InputSchema = z.object({
  messageId: z.string().uuid(),
  newText: z.string().min(1).max(2000),
});

export async function editAndSend(args: unknown) {
  const parsed = InputSchema.safeParse(args);
  if (!parsed.success) return { ok: false as const, error: "validation" };
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session?.user) return { ok: false as const, error: "unauthorized" };
  const msg = await findMessageById(parsed.data.messageId);
  if (!msg) return { ok: false as const, error: "not_found" };
  const owners = await findByUserId(session.user.id);
  if (!owners.find((o) => o.storeId === msg.storeId))
    return { ok: false as const, error: "forbidden" };
  if (msg.draftStatus !== "pending_review")
    return { ok: false as const, error: "invalid_state" };
  await markDraftEdited({ messageId: msg.id, newText: parsed.data.newText });
  const sendResult = await sendOutboundMessage({ messageId: msg.id });
  if (!sendResult.ok) return { ok: false as const, error: "send_failed" };
  await updateDraftStatus({
    messageId: msg.id,
    nextStatus: "sent",
    reviewerId: session.user.id,
  });
  revalidatePath("/store/inbox");
  return { ok: true as const };
}
```
